<script>
const canvas = document.getElementById("c");
const ctx = canvas.getContext("2d");
canvas.width = window.innerWidth;
canvas.height = window.innerHeight;

let stage = 0;
let progress = 0;
const particles = [];
const stars = [];

// Жүрек формуласы
function heart(t) {
    return { 
        x: 16 * Math.pow(Math.sin(t), 3), 
        y: -(13 * Math.cos(t) - 5 * Math.cos(2 * t) - 2 * Math.cos(3 * t) - Math.cos(4 * t)) 
    };
}

// Жұлдыздарды жасау
for(let i=0; i<200; i++) {
    stars.push({
        x: Math.random() * canvas.width,
        y: Math.random() * canvas.height,
        size: Math.random() * 1.5,
        blink: Math.random() * 0.05
    });
}

// Жүрек бөлшектерін жасау
for(let i=0; i<600; i++) {
    let t = Math.random() * Math.PI * 2;
    let h = heart(t);
    particles.push({
        tx: canvas.width/2 + h.x * 15,
        ty: canvas.height/2 + h.y * 15,
        x: Math.random() * canvas.width,
        y: Math.random() * canvas.height,
        size: Math.random() * 2 + 0.5,
        color: i % 2 === 0 ? "#f73487" : "#ff85a1",
        vx: Math.random() * 2 - 1,
        vy: Math.random() * 2 - 1
    });
}

function typeText() {
    let text = "Happy Birthday, Айназ 💖";
    let index = 0;
    let el = document.getElementById("msg");
    function write() {
        if(index < text.length) {
            el.innerHTML += text[index++];
            setTimeout(write, 100);
        }
    }
    write();
}

function draw() {
    // Аспан фонын жаңарту
    ctx.fillStyle = "rgba(9, 10, 15, 0.2)";
    ctx.fillRect(0, 0, canvas.width, canvas.height);

    // Жыпылықтаған жұлдыздар
    stars.forEach(s => {
        ctx.fillStyle = `rgba(255, 255, 255, ${0.5 + Math.sin(Date.now() * 0.001 + s.x) * 0.5})`;
        ctx.beginPath();
        ctx.arc(s.x, s.y, s.size, 0, Math.PI * 2);
        ctx.fill();
    });

    if (stage === 0) {
        particles.forEach(p => {
            // Жүрекке қарай ағу эффекті
            p.x += (p.tx - p.x) * 0.03;
            p.y += (p.ty - p.y) * 0.03;
            
            ctx.fillStyle = p.color;
            ctx.beginPath();
            ctx.arc(p.x, p.y, p.size, 0, Math.PI * 2);
            ctx.fill();
            
            // Жарқырау эффекті
            ctx.shadowBlur = 5;
            ctx.shadowColor = "#f73487";
        });

        progress++;
        if (progress === 180) {
            document.getElementById("text").style.opacity = 1;
            typeText();
            setTimeout(() => { stage = 1; }, 4000);
        }
    } else {
        // Жұлдызды жауын эффекті
        ctx.shadowBlur = 0;
        particles.forEach(p => {
            p.x += p.vx;
            p.y += p.vy + 0.5; // Төмен баяу құлау
            if(p.y > canvas.height) p.y = -10;

            ctx.fillStyle = p.color;
            ctx.beginPath();
            ctx.arc(p.x, p.y, p.size * 0.8, 0, Math.PI * 2);
            ctx.fill();
        });
        document.getElementById("text").classList.add("faded");
    }

    requestAnimationFrame(draw);
}

draw();

window.onresize = () => {
    canvas.width = window.innerWidth;
    canvas.height = window.innerHeight;
};
</script>
